class CustomStack {
    private int[] st;
    private int len;
    public CustomStack(int maxSize) {
        st = new int[maxSize];
        len = 0;
    }
    
    public void push(int x) {
        if(len < st.length){
            st[len++] = x;
        }
    }
    
    public int pop() {
        if(len > 0){
            len --;
            return st[len];
        }
        else 
            return -1;
    }
    
    public void increment(int k, int val) {
        if(k >= len) k = len;
        for(int i = 0; i < k; i ++){
            st[i] += val;
        }
    }
}

 * Your CustomStack object will be instantiated and called as such:
 * CustomStack obj = new CustomStack(maxSize);
 * obj.push(x);
 * int param_2 = obj.pop();
 * obj.increment(k,val);
 */
