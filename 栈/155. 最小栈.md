class MinStack {
    private Stack<Integer> st;
    private Stack<Integer> st_min;

    public MinStack() {
        st = new Stack<>();
        st_min = new Stack<>();
    }
    
    public void push(int val) {
        st.push(val);
        if (st_min.isEmpty() || val <= st_min.peek()) {
            st_min.push(val);
        }
    }
    
    public void pop() {
        int val = st.pop();
        if (val == st_min.peek()) {
            st_min.pop();
        }
    }
    
    public int top() {
        return st.peek();
    }
    
    public int getMin() {
        return st_min.peek();
    }
}
