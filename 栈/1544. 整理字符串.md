# [1544. 整理字符串](https://leetcode.cn/problems/make-the-string-great/)

```java

class Solution {
    public String makeGood(String s) {
        Stack<Character> st = new Stack<>();
        int diff = 'a' - 'A';
        for(char c : s.toCharArray()){
            if(st.empty()){
                st.push(c);
            }
            else{
                char b = st.peek();
                if(Math.abs(c-b) == diff){
                    st.pop();
                }
                else{
                    st.push(c);
                }
            }
        }
        StringBuilder sb = new StringBuilder();
        for (Character c : st) {
            sb.append(c);
        }
        return sb.toString();
    }
}
```
