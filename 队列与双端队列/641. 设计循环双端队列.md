# [641. 设计循环双端队列](https://leetcode.cn/problems/design-circular-deque/)

```java
class MyCircularDeque {
    private int maxLength;
    private int length;
    private int data[];
    private int posHead;
    private int posTail;
    public MyCircularDeque(int k) {
        data = new int[k];
        posHead = 0;
        posTail = k-1;
        length = 0;
        maxLength = k;
    }
    
    public boolean insertFront(int value) {
        if(length == this.maxLength){
            return false;
        }
        else{
            if(posHead == this.maxLength){
                posHead = 0;
            }
            data[posHead++] = value;
            length ++;
            return true;
        }
    }
    
    public boolean insertLast(int value) {
        if(length == this.maxLength){
            return false;
        }
        else{
            if(posTail == -1){
                posTail = this.maxLength - 1;
            }
            data[posTail--] = value;
            length ++;
            return true;
        }
    }
    
    public boolean deleteFront() {
        if(length == 0){
            return false;
        }
        else{
            posHead--;
            length --;
            if(posHead == -1){
                posHead = this.maxLength - 1;
            }
            data[posHead] = -1;
            return true;
        }
    }
    
    public boolean deleteLast() {
        if(length == 0){
            return false;
        }
        else{
            posTail++;
            length --;
            if(posTail == this.maxLength){
                posTail = 0;
            }
            data[posTail] = -1;
            return true;
        } 
    }
    
    public int getFront() {
        if(length == 0){
            return -1;
        }
        int index = posHead;
        if(index == 0){
            index = this.maxLength - 1;
        }
        else{
            index --;
        }
        return data[index];
    }
    
    public int getRear() {
        if(length == 0){
            return -1;
        }
        int index = posTail;
        if(index == this.maxLength - 1){
            index = 0;
        }
        else{
            index ++;
        }
        return data[index];
    }
    
    public boolean isEmpty() {
        if(this.length == 0){
            return true;
        }
        else{
            return false;
        }
    }
    
    public boolean isFull() {
        if(this.length == this.maxLength){
            return true;
        }
        else{
            return false;
        }
    }
}
```
