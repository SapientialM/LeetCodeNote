# [707. 设计链表](https://leetcode.cn/problems/design-linked-list/)

主要还是分界点的判断。

```java

class MyLinkedList {
    public class ListNode {
        // 结点的值
        int val;

        // 下一个结点
        ListNode next;

        // 节点的构造函数(无参)
        public ListNode() {
        }

        // 节点的构造函数(有一个参数)
        public ListNode(int val) {
            this.val = val;
        }

        // 节点的构造函数(有两个参数)
        public ListNode(int val, ListNode next) {
            this.val = val;
            this.next = next;
        }
    }
    
    private ListNode dummyNode;
    private int length;
    public MyLinkedList() {
        dummyNode = new ListNode(-1);
        length = 0;
    }
    
    public int get(int index) {
        if(index >= length) {
            return -1;
        }
        else{
            int pos = 0;
            ListNode node = dummyNode.next;
            while (pos != index) {
                pos ++;
                node = node.next;
            }
            return node.val;
        }
    }
    
    public void addAtHead(int val) {
        addAtIndex(0,val);
    }
    
    public void addAtTail(int val) {
        addAtIndex(length, val);
    }
    
    public void addAtIndex(int index, int val) {
        if(index < 0 || index > length) return;
        int pos = 0;
        ListNode node = dummyNode;
        while (pos != index) {
            pos ++;
            node = node.next;
        }
        if(node.next != null){
            ListNode newNode = new ListNode(val);
            newNode.next = node.next;
            node.next = newNode;
        }
        else{
            node.next = new ListNode(val);
        }
        length ++;
    }
    
    public void deleteAtIndex(int index) {
        if(index < 0 || index >= length) return;
        int pos = 0;
        ListNode node = dummyNode;
        while (pos != index) {
            pos ++;
            node = node.next;
        }
        if(node.next != null){
            node.next = node.next.next;
        }
        else{
            node.next = null;
        }
        length --;
    }
}
```

```java
class MyLinkedList {
    int size;
    ListNode head;

    public MyLinkedList() {
        size = 0;
        head = new ListNode(0);
    }

    public int get(int index) {
        if (index < 0 || index >= size) {
            return -1;
        }
        ListNode cur = head;
        for (int i = 0; i <= index; i++) {
            cur = cur.next;
        }
        return cur.val;
    }

    public void addAtHead(int val) {
        addAtIndex(0, val);
    }

    public void addAtTail(int val) {
        addAtIndex(size, val);
    }

    public void addAtIndex(int index, int val) {
        if (index > size) {
            return;
        }
        index = Math.max(0, index);
        size++;
        ListNode pred = head;
        for (int i = 0; i < index; i++) {
            pred = pred.next;
        }
        ListNode toAdd = new ListNode(val);
        toAdd.next = pred.next;
        pred.next = toAdd;
    }

    public void deleteAtIndex(int index) {
        if (index < 0 || index >= size) {
            return;
        }
        size--;
        ListNode pred = head;
        for (int i = 0; i < index; i++) {
            pred = pred.next;
        }
        pred.next = pred.next.next;
    }
}

class ListNode {
    int val;
    ListNode next;

    public ListNode(int val) {
        this.val = val;
    }
}
```
