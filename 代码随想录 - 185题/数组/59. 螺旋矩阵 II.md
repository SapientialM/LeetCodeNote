import javax.swing.text.html.parser.Element;

class Solution {
    public int[][] generateMatrix(int n) {
        int matrix[][] = new int[n][n];
        // 初始化
        for(int i = 0; i < n; i ++){
            for(int j = 0; j < n; j ++){
                matrix[i][j] = 0;
            }
        }
        int counter = 1;
        int i = 0, j = -1;
        // 0 1 2 3 右 下 左 上
        int direction = 0;
        while(counter <= n*n){
            if(direction == 0){
                j ++;
            }
            else if(direction == 1){
                i ++;
            }
            else if(direction == 2){
                j --;
            }
            else if(direction == 3){
                i --;
            }
            matrix[i][j] = counter ++;
            if(direction == 0 && j == n-1 || direction == 0 && matrix[i][j+1] != 0){
                direction = (direction + 1) % 4;
            }
            if(direction == 1 && i == n-1 || direction == 1 && matrix[i+1][j] != 0){
                direction = (direction + 1) % 4;
            }
            if(direction == 2 && j == 0 || direction == 2 && matrix[i][j-1] != 0){
                direction = (direction + 1) % 4;
            }
            if(direction == 3 && i == 0 || direction == 3 && matrix[i-1][j] != 0){
                direction = (direction + 1) % 4;
            }
        }
        return matrix;
    }
}
