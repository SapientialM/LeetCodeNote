class TextEditor {
    private StringBuilder content;
    private int cursor;

    public TextEditor() {
        content = new StringBuilder();
        cursor = 0;
    }

    public void addText(String text) {
        content.insert(cursor, text);
        cursor += text.length();
    }

    public int deleteText(int k) {
        int actualDelete = Math.min(k, cursor);
        content.delete(cursor - actualDelete, cursor);
        cursor -= actualDelete;
        return actualDelete;
    }

    public String cursorLeft(int k) {
        cursor = Math.max(0, cursor - k);
        int start = Math.max(0, cursor - 10);
        return content.substring(start, cursor);
    }

    public String cursorRight(int k) {
        cursor = Math.min(content.length(), cursor + k);
        int start = Math.max(0, cursor - 10);
        return content.substring(start, cursor);
    }
}
