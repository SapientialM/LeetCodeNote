class Solution {
    public ListNode swapNodes(ListNode head, int k) {
        ListNode pre,ne,fir,sec;
        pre = head;
        ne = head;
        int i = 1;
        while(i < k){
            ne = ne.next;
            i ++;
        }
        fir = ne;
        while(ne.next != null){
            pre = pre.next;
            ne = ne.next;
        }
        sec = pre;
        // exchange
        int temp = fir.val;
        fir.val = sec.val;
        sec.val = temp;
        return head;
    }
}
